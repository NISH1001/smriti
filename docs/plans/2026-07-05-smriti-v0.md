# Smriti v0 Prototype — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Prove the core loop on the real Pixel 9 — continuously buffer audio from a *selected* app, one tap captures `[−N, +(K−N)]` seconds, transcribe on-device with Whisper, and show the transcript as a notification.

**Architecture:** A foreground service holds a `MediaProjection` + `AudioPlaybackCapture` `AudioRecord` restricted to whitelisted app UIDs, feeding a fixed-size PCM `RingBuffer`. A floating overlay button triggers `SnipController`, which slices `[−N]` from the buffer and appends live audio to `+(K−N)`, then hands the finalized clip to a background `WhisperTranscriber` (whisper.cpp via JNI) whose text is posted as a notification. Pure logic (`RingBuffer`, `SnipController`) is JVM-unit-tested with TDD; platform pieces are verified on-device via `adb`/logcat.

**Tech Stack:** Kotlin + Jetpack Compose · foreground service · `MediaProjectionManager` / `AudioPlaybackCaptureConfiguration` · `SYSTEM_ALERT_WINDOW` overlay · whisper.cpp (ggml) via CMake `externalNativeBuild` + JNI · Whisper `tiny.en` Q5_1 model · CLI build (Gradle wrapper + `sdkmanager` + `adb`, no Android Studio).

- **Model:** whisper.cpp `ggml-tiny.en-q5_1.bin` (~31 MB) behind a `Transcriber` interface (`transcribe(pcm: FloatArray): String`) so the engine/model is swappable.

```kotlin
package com.nishparadox.smriti.capture

import android.app.*
import android.content.Intent
import android.content.pm.PackageManager
import android.media.*
import android.media.projection.MediaProjection
import android.media.projection.MediaProjectionManager
import android.os.Build
import android.util.Log
import androidx.core.app.NotificationCompat
import kotlin.concurrent.thread
import kotlin.math.sqrt

class ProbeCaptureService : Service() {
    override fun onBind(i: Intent?) = null

    override fun onStartCommand(intent: Intent, flags: Int, startId: Int): Int {
        val nm = getSystemService(NotificationManager::class.java)
        nm.createNotificationChannel(
            NotificationChannel("probe", "Probe", NotificationManager.IMPORTANCE_LOW))
        startForeground(1, NotificationCompat.Builder(this, "probe")
            .setContentTitle("Smriti probe capturing").setSmallIcon(android.R.drawable.ic_btn_speak_now).build())

        val resultCode = intent.getIntExtra("code", 0)
        val data: Intent = intent.getParcelableExtra("data")!!
        val mpm = getSystemService(MediaProjectionManager::class.java)
        val projection: MediaProjection = mpm.getMediaProjection(resultCode, data)

        // Restrict capture to the Audiobookshelf UID only.
        val absUid = packageManager.getPackageUid("com.audiobookshelf.app", 0)
        val config = AudioPlaybackCaptureConfiguration.Builder(projection)
            .addMatchingUsage(AudioAttributes.USAGE_MEDIA)
            .addMatchingUid(absUid)
            .build()

        val format = AudioFormat.Builder()
            .setEncoding(AudioFormat.ENCODING_PCM_16BIT)
            .setSampleRate(16000)
            .setChannelMask(AudioFormat.CHANNEL_IN_MONO)
            .build()
        val minBuf = AudioRecord.getMinBufferSize(16000,
            AudioFormat.CHANNEL_IN_MONO, AudioFormat.ENCODING_PCM_16BIT)
        val record = AudioRecord.Builder()
            .setAudioFormat(format)
            .setBufferSizeInBytes(minBuf * 2)
            .setAudioPlaybackCaptureConfig(config)
            .build()

        record.startRecording()
        thread {
            val buf = ShortArray(minBuf)
            repeat(200) {
                val n = record.read(buf, 0, buf.size)
                if (n > 0) {
                    var sum = 0.0
                    for (i in 0 until n) sum += buf[i].toDouble() * buf[i]
                    Log.i("SMRITI_PROBE", "rms=${sqrt(sum / n)} uid=$absUid n=$n")
                }
            }
            record.stop(); record.release(); projection.stop(); stopSelf()
        }
        return START_NOT_STICKY
    }
}
```

```kotlin
package com.nishparadox.smriti.capture

class RingBuffer(private val capacitySamples: Int) {
    private val data = ShortArray(capacitySamples)
    private var writePos = 0
    private var filled = 0

    @Synchronized fun write(src: ShortArray, len: Int) {
        for (i in 0 until len) {
            data[writePos] = src[i]
            writePos = (writePos + 1) % capacitySamples
            if (filled < capacitySamples) filled++
        }
    }

    @Synchronized fun snapshot(): ShortArray {
        val out = ShortArray(filled)
        val start = if (filled < capacitySamples) 0 else writePos
        for (i in 0 until filled) out[i] = data[(start + i) % capacitySamples]
        return out
    }
}
```

```kotlin
package com.nishparadox.smriti.capture

class SnipController(private val sampleRate: Int) {
    private var target = 0
    private val acc = ArrayList<Short>()

    fun begin(preRoll: ShortArray, totalSeconds: Int) {
        acc.clear()
        for (s in preRoll) acc.add(s)
        target = totalSeconds * sampleRate
    }

    /** Returns true once the clip has reached [target] samples. */
    fun feed(chunk: ShortArray, len: Int): Boolean {
        var i = 0
        while (i < len && acc.size < target) { acc.add(chunk[i]); i++ }
        return acc.size >= target
    }

    fun result(): ShortArray = acc.toShortArray()
}
```

`WavWriter.kt`:
```kotlin
package com.nishparadox.smriti.capture

import java.io.File
import java.io.DataOutputStream
import java.io.FileOutputStream
import java.nio.ByteBuffer
import java.nio.ByteOrder

object WavWriter {
    fun write(pcm: ShortArray, sampleRate: Int, file: File) {
        val dataBytes = pcm.size * 2
        DataOutputStream(FileOutputStream(file)).use { out ->
            fun le(vararg b: Int) = b.forEach { out.writeByte(it) }
            out.writeBytes("RIFF"); writeIntLE(out, 36 + dataBytes); out.writeBytes("WAVE")
            out.writeBytes("fmt "); writeIntLE(out, 16)
            writeShortLE(out, 1); writeShortLE(out, 1)                 // PCM, mono
            writeIntLE(out, sampleRate); writeIntLE(out, sampleRate * 2)
            writeShortLE(out, 2); writeShortLE(out, 16)                // block align, bits
            out.writeBytes("data"); writeIntLE(out, dataBytes)
            val bb = ByteBuffer.allocate(dataBytes).order(ByteOrder.LITTLE_ENDIAN)
            for (s in pcm) bb.putShort(s)
            out.write(bb.array())
        }
    }
    private fun writeIntLE(o: DataOutputStream, v: Int) {
        o.writeByte(v); o.writeByte(v shr 8); o.writeByte(v shr 16); o.writeByte(v shr 24)
    }
    private fun writeShortLE(o: DataOutputStream, v: Int) { o.writeByte(v); o.writeByte(v shr 8) }
}
```

```kotlin
package com.nishparadox.smriti.capture

import android.app.*
import android.content.Intent
import android.media.*
import android.media.projection.MediaProjection
import android.media.projection.MediaProjectionManager
import android.os.IBinder
import android.util.Log
import androidx.core.app.NotificationCompat
import com.nishparadox.smriti.trigger.SnipEntryPoint
import java.io.File
import java.util.concurrent.atomic.AtomicBoolean
import kotlin.concurrent.thread

class CaptureService : Service(), SnipEntryPoint {
    companion object { @Volatile var instance: CaptureService? = null }

    private val SR = 16000
    private var preRoll = 5
    private var total = 20
    private lateinit var ring: RingBuffer
    private lateinit var record: AudioRecord
    private lateinit var projection: MediaProjection
    private val running = AtomicBoolean(false)
    @Volatile private var snip: SnipController? = null

    override fun onBind(i: Intent?): IBinder? = null

    override fun onStartCommand(intent: Intent, flags: Int, startId: Int): Int {
        preRoll = intent.getIntExtra("preRoll", 5)
        total = intent.getIntExtra("total", 20)
        val uids = intent.getIntArrayExtra("uids") ?: intArrayOf()
        ring = RingBuffer(SR * preRoll)

        val nm = getSystemService(NotificationManager::class.java)
        nm.createNotificationChannel(NotificationChannel("cap", "Capture", NotificationManager.IMPORTANCE_LOW))
        startForeground(1, NotificationCompat.Builder(this, "cap")
            .setContentTitle("Smriti listening").setSmallIcon(android.R.drawable.ic_btn_speak_now).build())

        val mpm = getSystemService(MediaProjectionManager::class.java)
        projection = mpm.getMediaProjection(intent.getIntExtra("code", 0), intent.getParcelableExtra("data")!!)

        val cb = AudioPlaybackCaptureConfiguration.Builder(projection)
            .addMatchingUsage(AudioAttributes.USAGE_MEDIA)
        for (u in uids) cb.addMatchingUid(u)
        val config = cb.build()

        val fmt = AudioFormat.Builder().setEncoding(AudioFormat.ENCODING_PCM_16BIT)
            .setSampleRate(SR).setChannelMask(AudioFormat.CHANNEL_IN_MONO).build()
        val minBuf = AudioRecord.getMinBufferSize(SR, AudioFormat.CHANNEL_IN_MONO, AudioFormat.ENCODING_PCM_16BIT)
        record = AudioRecord.Builder().setAudioFormat(fmt)
            .setBufferSizeInBytes(minBuf * 2).setAudioPlaybackCaptureConfig(config).build()

        instance = this
        running.set(true)
        record.startRecording()
        thread {
            val buf = ShortArray(minBuf)
            while (running.get()) {
                val n = record.read(buf, 0, buf.size)
                if (n > 0) {
                    ring.write(buf, n)
                    val s = snip
                    if (s != null && s.feed(buf, n)) {
                        val clip = s.result(); snip = null
                        val dir = File(filesDir, "clips").apply { mkdirs() }
                        val f = File(dir, "${System.nanoTime()}.wav")
                        WavWriter.write(clip, SR, f)
                        Log.i("SMRITI_CAP", "saved ${f.absolutePath} samples=${clip.size}")
                    }
                }
            }
            record.stop(); record.release(); projection.stop()
        }
        return START_STICKY
    }

    override fun requestSnip() {
        if (snip == null) {
            val sc = SnipController(SR)
            sc.begin(ring.snapshot(), total)
            snip = sc
            Log.i("SMRITI_CAP", "snip started preRoll=$preRoll total=$total")
        }
    }

    override fun onDestroy() { running.set(false); instance = null; super.onDestroy() }
}
```
